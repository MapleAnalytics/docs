If you don't have a Next.js app yet, you can quickly create one with the following command:

```bash
npm create next-app@latest
```

<Steps titleSize="h3">
<Step  title="Install">

Now install the `maple-sdk` package

<Tabs>
  <Tab title="npm">
  ```bash
    npm install maple-sdk
    ```
  </Tab>
  <Tab title="pnpm">
  ```bash
    pnpm add maple-sdk
    ```
  </Tab>
  <Tab title="yarn">
  ```bash
    yarn add maple-sdk
    ```
  </Tab>
  <Tab title="bun">
  ```bash
    bun install maple-sdk
    ```
  </Tab>
</Tabs>
</Step>

<Step  title="Intializing Safesight">

Add the provider in your apps root layout, 
with your previously created websites token.

```ts /app/layout.ts
import { Analytics } from 'maple-sdk/react'
 
export default function RootLayout({ children }: { children: React.ReactNode }) {
	return (
		<html lang='en'>
			<head>
				<title>Next.js tracked with Maple</title>
			</head>
			<body>
				{children}
				<Analytics token={'yourWebsiteID'} />
			</body>
		</html>
	)
}
```

</Step>

<Step title="Running your NextJS app">

<Tabs>
  <Tab title="bun">
  ```bash
    bun run dev
    ```
  </Tab>
  <Tab title="pnpm">
  ```bash
    pnpm run dev
    ```
  </Tab>
  <Tab title="yarn">
  ```bash
    yarn run dev
    ```
  </Tab>
  <Tab title="npm">
  ```bash
    npm run dev
    ```
  </Tab>
</Tabs>
</Step>

<Step  title="Thats it!">

Go to your web console and you will see debug logs showing you some page views logs.
And once deployed data will come in!

</Step>
